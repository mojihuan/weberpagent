MAX_FILE_SIZE = 5 * 1024 * 1024  # 5MB

async def _validate_upload_file(file: UploadFile) -> bytes:
    """Validate upload file format and size. Returns file content bytes."""
    if not file.filename or not file.filename.endswith(".xlsx"):
        raise HTTPException(status_code=400, detail="仅支持 .xlsx 格式文件")
    content = await file.read()
    if len(content) == 0:
        raise HTTPException(status_code=400, detail="文件为空")
    if len(content) > MAX_FILE_SIZE:
        raise HTTPException(status_code=400, detail="文件大小不能超过 5MB")
    return content
```

@router.post("/import/preview")
async def import_preview(file: UploadFile = File(...)):
    content = await _validate_upload_file(file)
    result = parse_excel(BytesIO(content))
    return {
        "rows": [
            {
                "row_number": row.row_number,
                "data": row.data,
                "errors": row.errors,
                "valid": len(row.errors) == 0,
            }

        ],
        "total_rows": result.total_rows,
        "valid_count": sum(1 for r in result.rows if not r.errors),
        "error_count": sum(1 for r in result.rows if r.errors),
        "has_errors": result.has_errors,
    }
```

Add CONFIRM endpoint (per D-08, D-10, D-11). CRITICAL: Use `async with db.begin()` for atomic batch. Do NOT call TaskRepository.create() which commits per-row. Instead, replicate the serialization logic inline and use `db.add(task)`:
```python
@router.post("/import/confirm")
async def import_confirm(file: UploadFile = File(...), db: AsyncSession = Depends(get_db)):
    content = await _validate_upload_file(file)
    result = parse_excel(BytesIO(content))

    if result.has_errors:
        raise HTTPException(status_code=400, detail="文件包含无效行，无法导入。请返回检查数据")

    created_count = 0
    async with db.begin():
        for row in result.rows:
            task_data = dict(row.data)  # Copy because we mutate
            # Serialize preconditions to JSON string (same as TaskRepository.create)
            preconditions = task_data.get("preconditions")
            if preconditions is not None:
                task_data["preconditions"] = json_module.dumps(preconditions, ensure_ascii=False)
            # Map "assertions" key to "external_assertions" column (same as TaskRepository.create)
            assertions = task_data.pop("assertions", None)
            if assertions is not None:
                task_data["external_assertions"] = json_module.dumps(assertions, ensure_ascii=False)
            task = Task(**task_data, status="draft")
            db.add(task)
            created_count += 1
    # Transaction commits on block exit. Any exception rolls back all.

    return {"status": "success", "created_count": created_count}
```

IMPORTANT notes:
- The `import json as json_module` alias avoids any name collision with fastapi or starlette imports.
- `dict(row.data)` creates a mutable copy because ParsedRow.data comes from a frozen dataclass.
- `row.data` keys are: "name", "description", "target_url", "max_steps", "preconditions", "assertions" -- matching TEMPLATE_COLUMNS keys.
- The assertions key must be popped and renamed to external_assertions because the Task model has `external_assertions` as the column name, while `assertions` is a relationship field.
- Status defaults to "draft" per D-11 (imported tasks match manual creation).
- Backend validates file format/size per D-12 (defensive, not trusting frontend).
