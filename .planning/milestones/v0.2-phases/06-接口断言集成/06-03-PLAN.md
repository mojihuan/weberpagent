class TestTimeAssertion:
    """时间断言测试"""

    def test_time_assertion_now(self, service):
        """当前时间在范围内"""
        now = datetime.now()
        passed, msg = service.check_time_within_range(now)
        assert passed is True
        assert msg == ""

    def test_time_assertion_30_seconds_ago(self, service):
        """30秒前在范围内"""
        time_30s_ago = datetime.now() - timedelta(seconds=30)
        passed, msg = service.check_time_within_range(time_30s_ago)
        assert passed is True
        assert msg == ""

    def test_time_assertion_59_seconds_ago(self, service):
        """59秒前在范围内（边界条件）"""
        time_59s_ago = datetime.now() - timedelta(seconds=59)
        passed, msg = service.check_time_within_range(time_59s_ago)
        assert passed is True
        assert msg == ""

    def test_time_assertion_60_seconds_ago(self, service):
        """60秒前在范围内（边界条件）"""
        time_60s_ago = datetime.now() - timedelta(seconds=60)
        passed, msg = service.check_time_within_range(time_60s_ago)
        assert passed is True
        assert msg == ""

    def test_time_assertion_61_seconds_ago(self, service):
        """61秒前超出范围"""
        time_61s_ago = datetime.now() - timedelta(seconds=61)
        passed, msg = service.check_time_within_range(time_61s_ago)
        assert passed is False
        assert "超出容差范围" in msg
        assert "60 秒" in msg

    def test_time_assertion_string_format_standard(self, service):

        assert msg == ""

        assert passed is False
        assert "无法解析时间格式" in msg

        """自定义容差"""
        time_30s_ago = datetime.now() - timedelta(seconds=30)
        # 容差 20 秒，30 秒应该超出

        assert passed is False
        assert "20 秒" in msg
