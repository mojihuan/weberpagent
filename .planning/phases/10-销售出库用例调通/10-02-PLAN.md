```python
def sf_waybill() -> str:
    """Generate SF waybill number: "SF" + 12 hex chars"""
    return f"SF{uuid.uuid4().hex[:12].upper()}"

def random_phone() -> str:
    """Generate 11-digit phone: "13" + 9 random digits"""
    return "13" + ''.join(random.choices('0123456789', k=9))

def random_imei() -> str:
    """Generate 15-digit IMEI: "I" + 14 random digits"""
    return "I" + ''.join(random.choices('0123456789', k=14))

def random_serial() -> str:
    """Generate 8-digit serial number"""
    return ''.join(random.choices('0123456789', k=8))

def random_numbers(n: int) -> str:
    """Generate n-digit number string"""
    return ''.join(random.choices('0123456789', k=n))
```

```python
@staticmethod
def substitute_variables(text: str, context: dict[str, Any]) -> str:
    """Replace {{variable_name}} with context values using Jinja2 StrictUndefined"""
    env = Environment(
        variable_start_string='{{',
        variable_end_string='}}',
        undefined=StrictUndefined,  # Raises error if undefined
    )
    template = env.from_string(text)
    return template.render(**context)
```
