class TestBackwardCompatibility:
    """Tests for backward compatibility with legacy params parameter."""

    @pytest.mark.asyncio
    async def test_params_used_as_field_params_fallback(self):
        """execute_assertion_method uses params as field_params when field_params not provided."""
        mock_class = MagicMock
        mock_instance = MagicMock()
        mock_instance.test_method = MagicMock(return_value=None)
        mock_class.return_value = mock_instance

        with patch('backend.core.external_precondition_bridge.load_base_assertions_class') as mock_load:
            with patch('backend.core.external_precondition_bridge.resolve_headers') as mock_resolve:
                mock_load.return_value = ({'PcAssert': mock_class}, None)
                mock_resolve.return_value = {'Authorization': 'Bearer token'}

                # Call with legacy params (not field_params)
                result = await execute_assertion_method(
                    'PcAssert',
                    'test_method',
                    headers='main',
                    data='main',
                    params={'statusStr': '已完成'}  # Legacy param
                )

                assert result['success'] is True
                assert result['passed'] is True

    @pytest.mark.asyncio
    async def test_field_params_takes_precedence_over_params(self):
        """execute_assertion_method uses field_params when both field_params and params provided."""
        mock_class = MagicMock
        mock_instance = MagicMock()
        mock_instance.test_method = MagicMock(return_value=None)
        mock_class.return_value = mock_instance

        with patch('backend.core.external_precondition_bridge.load_base_assertions_class') as mock_load:

                # Call with both field_params and params

                    'PcAssert',
