    Replace the current publish() body (lines 37-51) with:
    ```python
    async def publish(self, run_id: str, event: str | None) -> None:
        """Publish event to all subscribers for a run.

        Per-queue exception isolation: a failing queue does not prevent
        other subscribers from receiving the event.
        """
        if event is not None:
            self._events[run_id].append(event)

        for queue in self._subscribers.get(run_id, []):
            try:
                await queue.put(event)
            except Exception:
                # Isolated: one bad queue must not break others
                logger.warning(f"[{run_id}] Failed to put event to subscriber queue, skipping")
    ```

    ```python
    async def check_element_exists(
        self, history: Any, selector: str
    ) -> tuple[bool, str, str]:
        """Check if a DOM element exists via CSS selector using page.evaluate.

        Per D-02: Uses history.page.evaluate() + document.querySelector(selector)
        consistent with project CDP interaction pattern.
        Per D-03: Returns (False, error_message, "") on detection failure.

        Args:
            history: browser-use execution history object (must have .page)
            selector: CSS selector string

        Returns:
            tuple[bool, str, str]: (passed, message, actual_value)
        """
        try:
            page = getattr(history, "page", None)
            if page is None:
                return False, f"Page not available for element check '{selector}'", ""

            result = await page.evaluate(
                f"JSON.stringify({{ found: document.querySelector('{selector}') !== null }})"
            )
            import json
            parsed = json.loads(result) if isinstance(result, str) else result
            if parsed.get("found"):
                return True, "", selector
            return False, f"Element '{selector}' not found in DOM", ""

        except Exception as e:
            return False, f"Element check failed for '{selector}': {str(e)}", ""
    ```
