from backend.agent.stall_detector import StallDetector

logger = logging.getLogger(__name__)

class MonitoredAgent(Agent):
    def __init__(
        self,
        *,
        stall_detector: StallDetector | None = None,
        pre_submit_guard: PreSubmitGuard | None = None,
        task_progress_tracker: TaskProgressTracker | None = None,
        **kwargs,
    ):
        super().__init__(**kwargs)
        self._stall_detector = stall_detector or StallDetector()
        self._pre_submit_guard = pre_submit_guard or PreSubmitGuard()
        self._task_tracker = task_progress_tracker or TaskProgressTracker()
        self._pending_interventions: list[str] = []

        # Parse task for progress tracking
        try:
            self._task_tracker.parse_task(self.task)
        except Exception:
            logger.warning("Failed to parse task for progress tracking")

`_prepare_context()` override (SUB-01):
```python
async def _prepare_context(self, step_info=None):
    result = await super()._prepare_context(step_info)
    # After super(), context_messages contain built-in nudges.
    # Inject our custom interventions.
    try:
        if self._pending_interventions:
            for msg in self._pending_interventions:
                self._message_manager._add_context_message(UserMessage(content=msg))
            self._pending_interventions = []
    except Exception as e:
        logger.error(f"[monitor] Failed to inject interventions: {e}")
    return result
```

`_execute_actions()` override (SUB-03):
```python
async def _execute_actions(self):
    if self.state.last_model_output is None:
        await super()._execute_actions()
        return

    actions = self.state.last_model_output.action
    if not actions:
        await super()._execute_actions()
        return

    # Check first action for submit intent
    first_action = actions[0]
    action_data = first_action.model_dump(exclude_unset=True)
    action_name = next(iter(action_data.keys()), 'unknown')

    if action_name == 'click' and self._pre_submit_guard:
        try:
            guard_result = self._pre_submit_guard.check(
                action_name=action_name,
                target_index=action_data.get('click', {}).get('index'),
                task=self.task,
                actual_values=None,  # Will be provided in Phase 50 integration
                submit_button_text=None,  # Will be provided in Phase 50 integration
            )
            if guard_result.should_block:
                # Block the submit click
                self.state.last_result = [ActionResult(error=guard_result.message)]
                self._message_manager._add_context_message(
                    UserMessage(content=guard_result.message)
                )
                logger.warning(f"[PreSubmitGuard] Blocked submit: {guard_result.message[:100]}")
                return
        except Exception as e:
            logger.error(f"[monitor] PreSubmitGuard check failed: {e}")

    await super()._execute_actions()
```

`create_step_callback()` method (SUB-02):
Returns an async callback for `register_new_step_callback` that:
1. Extracts action_name, target_index, evaluation from browser_state and agent_output
2. Computes dom_hash from DOM state
3. Calls stall_detector.check() -> if should_intervene, append message to _pending_interventions
4. Calls task_tracker.check_progress() -> if should_warn, append message to _pending_interventions
5. Calls task_tracker.update_from_evaluation() to mark completed steps
6. Wraps everything in try/except for fault tolerance (D-07/D-08)

```python
def create_step_callback(self) -> Callable:
    async def step_callback(browser_state, agent_output, step: int):
        try:
            # Extract data from callback parameters
            action_name = ""
            target_index = None
            evaluation = ""
            dom_hash = ""

            if agent_output and hasattr(agent_output, 'action') and agent_output.action:
                first_action = agent_output.action[0]
                action_dict = first_action.model_dump(exclude_none=True, mode='json')
                if action_dict:
                    action_name = list(action_dict.keys())[0]
                    params = action_dict[action_name]
                    target_index = params.get('index') if isinstance(params, dict) else None

            if agent_output and hasattr(agent_output, 'evaluation_previous_goal'):
                evaluation = agent_output.evaluation_previous_goal or ""

            if browser_state and hasattr(browser_state, 'dom_state'):
                dom_state = browser_state.dom_state
                if dom_state and hasattr(dom_state, 'llm_representation'):
                    try:
                        dom_str = dom_state.llm_representation()
                        import hashlib
                        dom_hash = hashlib.sha256(dom_str.encode('utf-8')).hexdigest()[:12]
                    except Exception:
                        dom_hash = ""

            # Stall detection
            stall_result = self._stall_detector.check(
                action_name=action_name,
                target_index=target_index,
                evaluation=evaluation,
                dom_hash=dom_hash,
            )
            if stall_result.should_intervene:
                self._pending_interventions.append(stall_result.message)
                logger.info(f"[monitor] Stall detected at step {step}: {stall_result.message[:80]}")

            # Progress tracking
            progress_result = self._task_tracker.check_progress(
                current_step=step,
                max_steps=getattr(self.state, 'max_steps', 30),
            )
            if progress_result.should_warn:
                self._pending_interventions.append(progress_result.message)
                logger.info(f"[monitor] Progress warning at step {step}: {progress_result.message[:80]}")

            self._task_tracker.update_from_evaluation(evaluation)

        except Exception as e:
            logger.error(f"[monitor] step_callback error (non-blocking): {e}")

    return step_callback
```

Key patterns:
- All detector calls wrapped in try/except (D-07/D-08: fault tolerant)
- _pending_interventions is the bridge between step_callback and _prepare_context
- _pending_interventions is cleared in _prepare_context AFTER injection, not in step_callback
- _execute_actions only checks first action for submit intent
- Pass actual_values=None in Phase 48 (will be wired in Phase 50)
