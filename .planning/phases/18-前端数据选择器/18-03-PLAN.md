       ```typescript
       import { useState } from 'react'
       import { ChevronRight, ChevronDown, Check } from 'lucide-react'

       export function JsonTreeViewer({
         data,
         onFieldClick,
         selectedPaths = [],
         maxDepth = 10
       }: JsonTreeViewerProps) {
         // Recursive component to render JSON nodes
         const renderNode = (value: any, path: string, depth: number): React.ReactNode => {
           if (depth > maxDepth) return <span className="text-gray-400">...</span>

           const isSelected = selectedPaths.includes(path)

           if (value === null) {
             return (
               <span
                 className={`text-gray-400 cursor-pointer ${isSelected ? 'bg-blue-50 px-1 rounded' : ''}`}
                 onClick={() => onFieldClick?.(path, null)}
               >
                 null {isSelected && <Check className="inline w-3 h-3 text-blue-500" />}
               </span>
             )
           }

           if (typeof value === 'boolean') {
             return (
               <span
                 className={`text-orange-600 cursor-pointer ${isSelected ? 'bg-blue-50 px-1 rounded' : ''}`}
                 onClick={() => onFieldClick?.(path, value)}
               >
                 {value.toString()} {isSelected && <Check className="inline w-3 h-3 text-blue-500" />}
               </span>
             )
           }

           if (typeof value === 'number') {
             return (
               <span
                 className={`text-blue-600 cursor-pointer ${isSelected ? 'bg-blue-50 px-1 rounded' : ''}`}
                 onClick={() => onFieldClick?.(path, value)}
               >
                 {value} {isSelected && <Check className="inline w-3 h-3 text-blue-500" />}
               </span>
             )
           }

           if (typeof value === 'string') {
             return (
               <span
                 className={`text-green-600 cursor-pointer ${isSelected ? 'bg-blue-50 px-1 rounded' : ''}`}
                 onClick={() => onFieldClick?.(path, value)}
               >
                 "{value}" {isSelected && <Check className="inline w-3 h-3 text-blue-500" />}
               </span>
             )
           }

           // ... continue for arrays and objects
         }

         return (
           <div className="font-mono text-sm overflow-auto max-h-96">
             {renderNode(data, '', 0)}
           </div>
         )
       }
       ```

       ```typescript
         setMethodConfigs(prev => {
           const next = new Map(prev)
           const config = next.get(key)
           if (config) {
             // Extract default variable name from path (last segment)
       ```

    3. Add code generation function:
       ```typescript
       const generateCode = (): string => {
         const lines: string[] = []
         methodConfigs.forEach(config => {
           const params = Object.entries(config.parameters)
             .map(([k, v]) => `${k}=${JSON.stringify(v)}`)
             .join(', ')
           const methodCall = `context.get_data('${config.methodName}', ${params})`

           config.extractions.forEach(ex => {
             // Convert path like "[0].imei" to Python accessor
             const accessor = ex.path
               .replace(/\[(\d+)\]/g, '[$1]')  // Keep array access as-is
               .replace(/\./g, "']['")         // Convert dot to bracket access
               .replace(/^\[/, '')             // Remove leading bracket
               .replace(/\]$/, '')             // Remove trailing bracket

             // Build the full accessor
             let fullAccessor: string
             if (ex.path.startsWith('[')) {
               // Path like [0].imei -> result[0]['imei']
               const parts = ex.path.split(/\.(.+)/)
               if (parts.length > 1) {
                 fullAccessor = `${methodCall}${parts[0]}['${parts[1].replace(/\./g, "']['")}']`
               } else {
                 fullAccessor = `${methodCall}${ex.path}`
               }
             } else {
               fullAccessor = `${methodCall}['${accessor}']`
             }

             // Simplified: direct path substitution
             const pathAccess = ex.path.replace(/\[(\d+)\]/g, '[$1]').replace(/\.([^.[]+)/g, "['$1']")
             lines.push(`${ex.variableName} = ${methodCall}${pathAccess}`)
           })
         })
         return lines.join('\n')
       }
       ```
