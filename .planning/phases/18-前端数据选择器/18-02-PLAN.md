const filteredModules = useMemo(() => {
  if (!searchQuery.trim()) return modules
  const query = searchQuery.toLowerCase()
  return modules

      ...module,
      operations: module.operations.filter(
        op => op.code.toLowerCase().includes(query) ||
              op.description.toLowerCase().includes(query)
      )
    }))
    .filter(module => module.operations.length > 0)
}, [modules, searchQuery])

    return next
  })
}

       // Create config entries for all selected methods
       selectedMethodKeys.forEach(key => {
         if (!methodConfigs.has(key)) {
           const [className, methodName] = key.split(':')
           // Find the method to get its parameters
           const cls = methods.classes.find(c => c.name === className)
           const method = cls?.methods.find(m => m.name === methodName)
           // Initialize params with defaults
           const params: Record<string, any> = {}
           method?.parameters.forEach(p => {
             if (p.default !== null) {
               // Parse default value based on type
               params[p.name] = p.type === 'int' ? parseInt(p.default) : p.default
             }
           })
           methodConfigs.set(key, {
             className,
             methodName,
             parameters: params,
             extractions: []
           })
         }
       })

    3. Add updateParameter function:
       ```typescript
       const updateParameter = (key: string, paramName: string, value: any) => {
         setMethodConfigs(prev => {
           const next = new Map(prev)
           const config = next.get(key)
           if (config) {
             next.set(key, {
               ...config,
               parameters: { ...config.parameters, [paramName]: value }
             })
           }
           return next
         })
       }
       ```
